S: INSTRUCCIONES

INSTRUCCIONES: $< INSTRUCCIONES2 >$

INSTRUCCIONES2: INSTRUCCIONES2 "," INSTRUCCION
	|INSTRUCCION

INSTRUCCION: DATABASES 
		| USERS

DATABASES: "DATABASES" = [DATABASES2]

DATABASES2: DATABASES2 "," DATABASE
			| DATABASE
			| EMPTY

DATABASE: < DATABASE2 >

DATABASE2: DATABASE2 , DATABASE3
	| DATABASE3

DATABASE3: NAME
	| DATA

NAME: "NAME" = VALOR

DATA: "DATA" = [DATA5]

DATA5: DATA5 , DATA2
	| DATA2
	
DATA2: <DATA3>

DATA3: DATA3 , DATA4
	|DATA4

DATA4: TABLA
	| OBJETO
	| PROCEDURE

TABLA: CQLTYPE
	| NAME
	| COLUMNS
	| DATA_DATA

CQLTYPE: "CQL-TYPE" = VALOR

COLUMNS: "COLUMNS" = [COLUMNS4]

COLUMNS4: COLUMNS4 , COLUMNS2
	| COLUMNS2

COLUMNS2: <COLUMNS3>

COLUMNS3: COLUMNS3 , COLUMN
	| COLUMN

COLUMN: NAME
	| TYPE
	| PK

TYPE: "TYPE" = VALOR

PK: "PK" = BOOL

DATA_DATA: "DATA" [ DATA_DATA2 ]

DATA_DATA2: ${valor.valor}$
		| DATA_DATA3
		| EMPTY

DATA_DATA3: DATA_DATA3 , DATA_DATA4
		| DATA_DATA4

DATA_DATA4: <DATA_DATA5>

DATA_DATA5: DATA_DATA5 ,  DATA_DATA6
		| DATA_DATA6

DATA_DATA6: VALOR = VALOR


OBJETO: CQL-TYPE
	| NAME
	| ATTRIBUTES

ATTRIBUTES: "ATTRS" = [ATTRIBUTES2]

ATTRIBUTES2: ATTRIBUTES2, ATTRIBUTES3
	| ATTRIBUTES3
	| EMPTY

ATTRIBUTES3: <ATTRIBUTES4>

ATTRIBUTES4: ATTRIBUTES4 , ATTRIBUTE
		| ATTRIBUTE

ATTRIBUTE: NAME
	| TYPE


PROCEDURES	: CQL-TYPE
			| NAME
			| PARAMETERS
			| INSTR

INSTR: "INSTR" =  CONT_DATA_IMPORT 

PARAMETERS	: "PARAMETERS" = [PARAMETERS2]

PARAMETERS2	: PARAMETERS2, PARAMETERS3
			| PARAMETERS3
			| EMPTY

PARAMETERS3	: <PARAMETERS4>

PARAMETERS4	: PARAMETERS4, PARAMETER
			| PARAMETER

PARAMETERS5: NAME
		| TYPE
		| AS

AS: "AS" = IN_OUT


USERS: "USERS" = [USERS2]

USERS2: USERS2 , USERS3
	| USERS3
	| EMPTY

USERS3: <USERS4>

USERS4: USERS4, USERS5
	| USERS5

USERS5: NAME
	| PASWORD
	| PERMISSIONS

PASSWORD: "PASSWORD" = VALOR

PERMISSIONS: "PERMISSIONS" = [PERMISSIONS2]

PERMISSIONS2: PERMISSIONS2 , PERMISSION
	| PERMISSION
	| EMPTY

PERMISSION: < NAME >
